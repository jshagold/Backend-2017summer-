CREATE TABLE user (
	_id INT NOT NULL AUTO_INCREMENT,
	name varchar(32),
	phone_number char(11),
	email varchar(32),
	password varchar(32),
	PRIMARY KEY (_id)
);

CREATE TABLE category (
	_id INT NOT NULL AUTO_INCREMENT,
	name varchar(20),
	PRIMARY KEY (_id)
);

create table post (
	_id int not null auto_increment,
	title varchar(32),
	content text(200),
	user_id int not null,
	foreign key(user_id)
	references user(_id),
	primary key (_id)
);

CREATE TABLE comment (
	_id int not null auto_increment,
	content text(50),
	user_id int not null,
	foreign key(user_id)
	references user(_id),
	primary key (_id)
);

create table post_comment (
	_id int not null auto_increment,
	post_id int not null,
	foreign key(post_id)
	references post(_id),
	comment_id int not null,
	foreign key(comment_id)
	references comment(_id),
	primary key (_id)
);

create table find_job (
	_id int not null auto_increment,
	category_id int not null,
	foreign key(category_id)
	references category(_id),
	user_id int not null,
	foreign key(user_id)
	references user(_id),
	title varchar(32) not null,
	content text(200),
	latitude double not null,
	longitute double not null,
	start_time varchar(9) not null,
	end_time varchar(9) not null,
	primary key (_id)
);

create table find_job_comment (
	_id int not null auto_increment,
	find_job_id int not null,
	foreign key(find_job_id)
	references find_job(_id),
	comment_id int not null,
	foreign key(comment_id)
	references comment(_id),
	primary key(_id)
);

create table vote (
	_id int not null auto_increment,
	find_job_id int not null,
	foreign key(find_job_id)
	references find_job(_id),
	primary key(_id)
);

create table tbl_select (
	_id int not null auto_increment,
	content varchar(50),
	vote_id int not null,
	foreign key(vote_id)
	references vote(_id),
	primary key(_id)
);

create table tbl_select_user (
	_id int not null auto_increment,
	tbl_select_id int not null,
	foreign key(tbl_select_id)
	references tbl_select(_id),
	user_id int not null,
	foreign key(user_id)
	references user(_id),
	primary key(_id)
);
